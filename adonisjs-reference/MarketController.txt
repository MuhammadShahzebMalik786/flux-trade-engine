/**

 * 
 * Location: app/Controllers/Http/MarketController.ts (or app/controllers/market_controller.ts in v6)
 * Description: Clean, enterprise-ready controller handling session authentications,
 * web asset delivery, and proxy routing to protect secure API keys.
 */

// If using AdonisJS v5 / v6
import { HttpContextContract } from '@ioc:Adonis/Core/HttpContext'
// For AdonisJS v6 standard decorators
// import { HttpContext } from '@adonisjs/core/http'

export default class MarketController {
  /**
   * Serve the primary dashboard view index
   * Renders the compiled SPA single-page document.
   */
  public async index({ view, auth }: HttpContextContract) {
    // Optional: Protect the workspace - assert that user is authenticated
    const user = auth.user
    
    return view.render('dashboard', {
      user: user ? { id: user.id, email: user.email } : null,
      serverTime: new Date().toISOString(),
      apiVersion: '1.0.0-gold'
    })
  }

  /**
   * Secure Endpoint Proxy: 获取 Binance API 的历史 K 线数据
   * Prevents revealing upstream endpoints directly to the client
   * and allows applying server-side caching if desired.
   */
  public async getKlines({ request, response }: HttpContextContract) {
    const symbol = request.input('symbol', 'BTCUSDT')
    const interval = request.input('interval', '1m')
    const limit = request.input('limit', '80')

    try {
      // Upstream fetch to Binance CoreREST
      const upstreamUrl = `https://api.binance.com/api/v3/klines?symbol=${symbol}&interval=${interval}&limit=${limit}`
      const fetchResponse = await fetch(upstreamUrl)
      
      if (!fetchResponse.ok) {
        return response.status(fetchResponse.status).send({
          error: 'Upstream connection error from Binance endpoints'
        })
      }

      const data = await fetchResponse.json()

      // Set proper headers for API integrity and caching
      response.header('Cache-Control', 'public, max-age=15') // 15s cache
      return response.status(200).send(data)

    } catch (error) {
      return response.status(500).send({
        error: 'Secured proxy failing to fetch aggregate candle data',
        details: error.message
      })
    }
  }

  /**
   * Secure WebSocket Authenticator Challenge (Ticket system)
   * Dispatches transient security tokens for real-time WebSocket client connection loops.
   */
  public async authorizeStream({ auth, response }: HttpContextContract) {
    try {
      const user = await auth.use('web').authenticate()
      if (!user) {
        return response.status(401).send({ error: 'Unauthorized credentials' })
      }

      // Generate a secure single-use ticket/token stored in Redis/DB of short TTL
      const token = Math.random().toString(36).substring(2, 15)

      return response.status(200).send({
        ticket: token,
        wsEndpoint: 'wss://stream.binance.com:9443/ws/btcusdt@trade',
        expiresInSeconds: 60
      })
    } catch {
      return response.status(401).send({ error: 'Credentials challenge failed' })
    }
  }
}
